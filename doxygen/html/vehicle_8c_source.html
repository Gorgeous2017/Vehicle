153 void DisplayOneSMG(int p, int num, int dot)
154 {
155  status(p)
156  {
157  case 0:
158  rGPGDAT &= ~(1 << 2); /* 设置数码管的位选 GPG2 = 0 (COM1) */
159  rGPEDAT &= ~(1 << 9); /* 设置数码管的位选 GPE9 = 0 (COM2) */
160  break;
161  case 1:
162  rGPGDAT |= (1 << 2); /* 设置数码管的位选 GPG2 = 1 (COM1) */
163  rGPEDAT &= ~(1 << 9); /* 设置数码管的位选 GPE9 = 0 (COM2) */
164  break;
165  case 2:
166  rGPGDAT &= ~(1 << 2); /* 设置数码管的位选 GPG2 = 0 (COM1) */
167  rGPEDAT |= (1 << 9); /* 设置数码管的位选 GPE9 = 1 (COM2) */
168  break;
169  case 3:
170  rGPGDAT |= (1 << 2); /* 设置数码管的位选 GPG2 = 1 (COM1) */
171  rGPEDAT |= (1 << 9); /* 设置数码管的位选 GPE9 = 1 (COM2) */
172  break;
173  }
174 
175  rGPEDAT = (rGPEDAT & ~(0x3C00)) | (num << 10); /* 对应数字 6 */
176  rGPHDAT = (rGPHDAT & ~(1 << 8)) | (dot << 8);
177 }
